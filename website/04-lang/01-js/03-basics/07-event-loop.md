function print(a, b, c, d) {
  console.log({ a, b, c, d });
}

function transform(fn, context = null) {
  let resultFn = fn;
  console.log('base len: ', fn.length);

  function stepFn(arg) {
    resultFn = resultFn.bind(context, arg);

    console.log({ len: resultFn.length, resultFn });

    if (resultFn.length) {
      return stepFn;
    }

    return resultFn();
  }

  return stepFn;
}
